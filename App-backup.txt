import React, { useState, useMemo } from 'react';
import { 
  Plus, 
  Minus, 
  Search, 
  AlertTriangle, 
  TrendingUp, 
  Layers, 
  DollarSign, 
  Trash2, 
  Edit3, 
  LogOut, 
  ShoppingBag,
  Filter
} from 'lucide-react';

export default function App() {
  // Mock active user session for our visual playground
  const [shopName] = useState("My Provisions Shop");
  
  // Local state initialized with clean dummy capstone presentation data
  const [inventory, setInventory] = useState([
    { id: '1', product_name: 'Milo Chocolate Malt 400g', sku: 'MILO-400', quantity: 14, threshold: 5, unit_price: 2800.00 },
    { id: '2', product_name: 'Peak Full Cream Milk Powder', sku: 'PEAK-MILK', quantity: 3, threshold: 8, unit_price: 3200.00 },
    { id: '3', product_name: 'Dangote Granulated Sugar 1kg', sku: 'SUG-DANG', quantity: 0, threshold: 10, unit_price: 1500.00 },
    { id: '4', product_name: 'Golden Penny Pasta 500g', sku: 'GP-PASTA', quantity: 45, threshold: 15, unit_price: 850.00 },
    { id: '5', product_name: 'Mama Gold Rice 5kg', sku: 'RICE-MG5', quantity: 2, threshold: 5, unit_price: 7500.00 },
  ]);

  const [searchTerm, setSearchTerm] = useState('');
  const [filterMode, setFilterMode] = useState('all'); // 'all' | 'alerts'

  // --- Dynamic Business Logic Analytics Engine (Calculates metrics from PRD Section 4.1) ---
  const stats = useMemo(() => {
    let totalItems = inventory.length;
    let lowStockCount = 0;
    let totalValue = 0;

    inventory.forEach(item => {
      totalValue += item.quantity * item.unit_price;
      if (item.quantity <= item.threshold) {
        lowStockCount++;
      }
    });

    return { totalItems, lowStockCount, totalValue };
  }, [inventory]);

  // Helper calculation matching mathematical rule inside PRD section 3.4
  const getItemStatus = (quantity, threshold) => {
    if (quantity === 0) return { label: 'Out of Stock', classes: 'bg-red-50 text-red-700 border-red-200' };
    if (quantity <= threshold) return { label: 'Low Stock', classes: 'bg-amber-50 text-amber-700 border-amber-200' };
    return { label: 'In Stock', classes: 'bg-emerald-50 text-emerald-700 border-emerald-200' };
  };

  // --- Quick Stock Inlines (PRD Section 3.3) ---
  const adjustStock = (id, change) => {
    setInventory(prev => prev.map(item => {
      if (item.id === id) {
        const targetQuantity = item.quantity + change;
        // Defensive barrier check (PRD Req 3.3.2)
        return { ...item, quantity: targetQuantity < 0 ? 0 : targetQuantity };
      }
      return item;
    }));
  };

  const deleteItem = (id) => {
    if(window.confirm("Are you sure you want to delete this product SKU?")) {
      setInventory(prev => prev.filter(item => item.id !== id));
    }
  };

  // Filtered Inventory List parsing search metrics
  const filteredInventory = useMemo(() => {
    return inventory.filter(item => {
      const matchesSearch = item.product_name.toLowerCase().includes(searchTerm.toLowerCase()) || 
                            item.sku.toLowerCase().includes(searchTerm.toLowerCase());
      const matchesFilter = filterMode === 'all' || item.quantity <= item.threshold;
      return matchesSearch && matchesFilter;
    });
  }, [inventory, searchTerm, filterMode]);

  return (
    <div className="min-h-screen bg-slate-50 font-sans text-slate-800 antialiased">
      
      {/* 1. TOP PREMIUM APP BAR */}
      <nav className="sticky top-0 z-50 border-b border-slate-200 bg-white px-6 py-4 shadow-sm">
        <div className="mx-auto flex max-w-7xl items-center justify-between">
          <div className="flex items-center space-x-3">
            <div className="flex h-10 w-10 items-center justify-center rounded-xl bg-blue-600 text-white shadow-md shadow-blue-200">
              <ShoppingBag className="h-5 w-5" />
            </div>
            <div>
              <span className="text-xl font-bold tracking-tight text-slate-900">StockPulse</span>
              <span className="ml-2 rounded-md bg-blue-50 px-2 py-0.5 text-xs font-semibold text-blue-700">MVP</span>
            </div>
          </div>
          
          <div className="flex items-center space-x-4">
            <div className="hidden text-right md:block">
              <p className="text-sm font-medium text-slate-900">{shopName}</p>
              <p className="text-xs text-slate-500">Merchant Terminal</p>
            </div>
            <button className="flex h-9 w-9 items-center justify-center rounded-lg border border-slate-200 text-slate-500 transition hover:bg-slate-50 hover:text-red-600">
              <LogOut className="h-4 w-4" />
            </button>
          </div>
        </div>
      </nav>

      <main className="mx-auto max-w-7xl p-4 sm:p-6 lg:p-8">
        
        {/* 2. DYNAMIC KPI ANALYTICS HERO BAR */}
        <div className="mb-8 grid grid-cols-1 gap-5 sm:grid-cols-3">
          {/* Card 1: Total SKUs */}
          <div className="relative overflow-hidden rounded-2xl border border-slate-200 bg-white p-6 shadow-sm transition hover:shadow-md">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-sm font-medium uppercase tracking-wider text-slate-500">Total Unique SKUs</p>
                <h3 className="mt-2 text-3xl font-bold text-slate-900">{stats.totalItems}</h3>
              </div>
              <div className="rounded-xl bg-slate-100 p-3 text-slate-600">
                <Layers className="h-6 w-6" />
              </div>
            </div>
            <div className="mt-4 flex items-center text-xs text-slate-500">
              <span className="font-medium text-slate-900">Active catalog items</span>
            </div>
          </div>

          {/* Card 2: Low Stock Warning Engine */}
          <div className={`relative overflow-hidden rounded-2xl border p-6 shadow-sm transition hover:shadow-md ${stats.lowStockCount > 0 ? 'border-amber-200 bg-white' : 'border-slate-200 bg-white'}`}>
            <div className="flex items-center justify-between">
              <div>
                <p className="text-sm font-medium uppercase tracking-wider text-slate-500">Alert Items</p>
                <h3 className={`mt-2 text-3xl font-bold ${stats.lowStockCount > 0 ? 'text-amber-600' : 'text-slate-900'}`}>
                  {stats.lowStockCount}
                </h3>
              </div>
              <div className={`rounded-xl p-3 ${stats.lowStockCount > 0 ? 'bg-amber-50 text-amber-600' : 'bg-slate-100 text-slate-600'}`}>
                <AlertTriangle className="h-6 w-6" />
              </div>
            </div>
            <div className="mt-4 flex items-center text-xs text-slate-500">
              <span className={`font-semibold ${stats.lowStockCount > 0 ? 'text-amber-600' : 'text-slate-600'}`}>
                {stats.lowStockCount > 0 ? 'Needs immediate reordering' : 'Stock levels healthy'}
              </span>
            </div>
          </div>

          {/* Card 3: Total Asset Valuation */}
          <div className="relative overflow-hidden rounded-2xl border border-slate-200 bg-white p-6 shadow-sm transition hover:shadow-md">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-sm font-medium uppercase tracking-wider text-slate-500">Total Stock Asset Value</p>
                <h3 className="mt-2 text-3xl font-bold text-slate-900">
                  ₦{stats.totalValue.toLocaleString('en-US', { minimumFractionDigits: 2 })}
                </h3>
              </div>
              <div className="rounded-xl bg-emerald-50 p-3 text-emerald-600">
                <DollarSign className="h-6 w-6" />
              </div>
            </div>
            <div className="mt-4 flex items-center text-xs text-emerald-600">
              <TrendingUp className="mr-1 h-3.5 w-3.5" />
              <span className="font-medium text-slate-500">Evaluated at current unit cost</span>
            </div>
          </div>
        </div>

        {/* 3. CONTROLS, FILTERS & ACTION REGION */}
        <div className="mb-6 flex flex-col gap-4 sm:flex-row sm:items-center sm:justify-between">
          
          {/* Real-time search entry form */}
          <div className="relative flex-1 max-w-md">
            <div className="pointer-events-none absolute inset-y-0 left-0 flex items-center pl-3 text-slate-400">
              <Search className="h-4 w-4" />
            </div>
            <input
              type="text"
              placeholder="Search product name or SKU..."
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
              className="w-full rounded-xl border border-slate-200 bg-white py-2.5 pl-10 pr-4 text-sm text-slate-900 outline-none transition focus:border-blue-500 focus:ring-2 focus:ring-blue-100"
            />
          </div>

          {/* Scope Filters (All Items vs Active Alert Toggles) */}
          <div className="flex items-center space-x-3">
            <div className="inline-flex rounded-xl bg-slate-200/70 p-1">
              <button
                onClick={() => setFilterMode('all')}
                className={`rounded-lg px-4 py-1.5 text-xs font-semibold transition ${filterMode === 'all' ? 'bg-white text-slate-900 shadow-sm' : 'text-slate-600 hover:text-slate-900'}`}
              >
                All Stock
              </button>
              <button
                onClick={() => setFilterMode('alerts')}
                className={`relative rounded-lg px-4 py-1.5 text-xs font-semibold transition ${filterMode === 'alerts' ? 'bg-white text-slate-900 shadow-sm' : 'text-slate-600 hover:text-slate-900'}`}
              >
                Alerts Only
                {stats.lowStockCount > 0 && (
                  <span className="absolute -top-1 -right-1 flex h-2.5 w-2.5 rounded-full bg-amber-500"></span>
                )}
              </button>
            </div>

            <button className="flex items-center space-x-2 rounded-xl bg-blue-600 px-4 py-2.5 text-sm font-semibold text-white shadow-sm transition hover:bg-blue-700 shadow-blue-100">
              <Plus className="h-4 w-4" />
              <span className="hidden sm:inline">Add Product</span>
            </button>
          </div>
        </div>

        {/* 4. MAIN DATA ARCHITECTURE SHELF */}
        <div className="overflow-hidden rounded-2xl border border-slate-200 bg-white shadow-sm">
          <div className="overflow-x-auto">
            <table className="w-full border-collapse text-left text-sm">
              <thead className="bg-slate-50 text-xs font-semibold uppercase tracking-wider text-slate-500 border-b border-slate-200">
                <tr>
                  <th className="px-6 py-4">Product Details</th>
                  <th className="px-6 py-4">SKU Code</th>
                  <th className="px-6 py-4">Unit Value</th>
                  <th className="px-6 py-4 text-center">Current Stock Counter</th>
                  <th className="px-6 py-4">Status Threshold</th>
                  <th className="px-6 py-4 text-right">Action Workspace</th>
                </tr>
              </thead>
              <tbody className="divide-y divide-slate-100 bg-white">
                {filteredInventory.length > 0 ? (
                  filteredInventory.map((item) => {
                    const status = getItemStatus(item.quantity, item.threshold);
                    return (
                      <tr key={item.id} className="transition hover:bg-slate-50/50">
                        {/* Title Info */}
                        <td className="px-6 py-4">
                          <p className="font-semibold text-slate-900">{item.product_name}</p>
                        </td>
                        
                        {/* SKU */}
                        <td className="px-6 py-4 text-slate-500 font-mono text-xs">
                          {item.sku || 'N/A'}
                        </td>
                        
                        {/* Unit Price */}
                        <td className="px-6 py-4 font-medium text-slate-900">
                          ₦{item.unit_price.toLocaleString('en-US')}
                        </td>
                        
                        {/* Fast Adjuster Counter */}
                        <td className="px-6 py-4">
                          <div className="flex items-center justify-center space-x-3">
                            <button
                              onClick={() => adjustStock(item.id, -1)}
                              disabled={item.quantity === 0}
                              className="flex h-8 w-8 items-center justify-center rounded-lg border border-slate-200 bg-white text-slate-600 transition hover:bg-slate-50 disabled:opacity-40 disabled:hover:bg-white"
                            >
                              <Minus className="h-3.5 w-3.5" />
                            </button>
                            <span className="w-8 text-center font-bold text-slate-900 text-base">
                              {item.quantity}
                            </span>
                            <button
                              onClick={() => adjustStock(item.id, 1)}
                              className="flex h-8 w-8 items-center justify-center rounded-lg border border-slate-200 bg-white text-slate-600 transition hover:bg-slate-50"
                            >
                              <Plus className="h-3.5 w-3.5" />
                            </button>
                          </div>
                        </td>
                        
                        {/* Status Condition Badge Block */}
                        <td className="px-6 py-4">
                          <span className={`inline-flex items-center rounded-full border px-2.5 py-1 text-xs font-medium tracking-wide ${status.classes}`}>
                            {status.label}
                          </span>
                        </td>
                        
                        {/* System Action Suite */}
                        <td className="px-6 py-4 text-right">
                          <div className="flex items-center justify-end space-x-2">
                            <button className="p-1.5 text-slate-400 hover:text-slate-600 transition rounded-lg hover:bg-slate-100">
                              <Edit3 className="h-4 w-4" />
                            </button>
                            <button 
                              onClick={() => deleteItem(item.id)}
                              className="p-1.5 text-slate-400 hover:text-red-600 transition rounded-lg hover:bg-red-50"
                            >
                              <Trash2 className="h-4 w-4" />
                            </button>
                          </div>
                        </td>
                      </tr>
                    );
                  })
                ) : (
                  <tr>
                    <td colSpan="6" className="px-6 py-12 text-center text-slate-400">
                      <div className="flex flex-col items-center justify-center">
                        <Filter className="h-8 w-8 text-slate-300 mb-2" />
                        <p className="text-sm">No inventory records matching your active filters.</p>
                      </div>
                    </td>
                  </tr>
                )}
              </tbody>
            </table>
          </div>
        </div>
        
      </main>
    </div>
  );
}
